    【例6.10】用公式求π。
    #include <math.h>
main()
{
int s;
float n,t,pi;
t=1,pi=0;n=1.0;s=1;
while(fabs(t)>1e-6)
  {pi=pi+t;
   n=n+2;  
   s=-s;
   t=s/n;
  }
pi=pi*4; 
printf("pi=%10.6f\n",pi);
}

    【例6.11】判断m是否素数。

    #include <math.h>
main()
{
int m,i,k;
scanf(“%d”,&m);
k=sqrt(m);
for(i=2;i<=k;i++)
   if(m%i==0)break;
   if(i>=k+1)
   printf(“%d is a prime number\n”,m);
   else
  printf(“%d is not a prime number\n”,m);
}
